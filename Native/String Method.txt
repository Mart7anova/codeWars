//1
const repeatString = (str, num, simbol) => {
    return (str + simbol).repeat(num).slice(0, -1)
}
console.log(repeatString("yo", 3, " "))
console.log(repeatString("yo", 3, ","))

//2
const checkStart = (text, str) => {
    const newText = text.toLowerCase().slice(0, str.length)
    return newText.includes(str.toLowerCase())
}
console.log(checkStart("Incubator", "inc"))
console.log(checkStart("Incubator", "yo"))

//3
const truncateString = (str, num) => {
    return str.slice(0, num) + "..."
}
console.log(truncateString("Всем студентам инкубатора желаю удачи!", 10))

//4
const getMinLengthWord = (str) => {
    return str.split(' ').reduce((acc, st)=> acc.length < st.length ? acc : st)
}
console.log(getMinLengthWord("Всем студентам инкубатора желаю удачи!"))
console.log(getMinLengthWord("Меня зовут Ли"))

//5
const setUpperCase = (str) => {
    return str.toLowerCase().split(' ').map(s => s[0].toUpperCase() + s.substring(1)).join(' ')
}
console.log(setUpperCase("всем стУдентам инкуБатора Желаю удачИ!"))

//6
const isIncludes = (text, str) => {
    const newStr = str.toLowerCase()
    const newText = text.toLowerCase().split('').filter((str)=> {
        for (let i = 0; i < newStr.length; i++) {
            if(newStr[i] === str) return str
        }
    } )
    return newText.length === str.length
}
console.log(isIncludes("Incubator", "Cut"))
console.log(isIncludes("Incubator", "table"))
